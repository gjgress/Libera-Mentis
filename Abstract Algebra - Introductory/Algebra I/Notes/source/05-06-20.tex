\begin{lemma}[Burnside's Lemma]
	The number of orbits is equal to the average number of fixed points. We can write this by
	\begin{align*}
		\left| G \right| \cdot \text{(number of orbits)} = \sum_{g\in G} \left| S^{g} \right| 
	\end{align*}
\end{lemma}
